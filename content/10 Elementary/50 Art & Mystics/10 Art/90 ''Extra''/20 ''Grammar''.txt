with %% preposition           = с, со
''Good luck '''with''' that.'' = ''Удачи тебе с этим.''
''I'm down '''with''' that.'' = ''Я завязал с этим.'' %% idiomatic

without %% preposition, adverb = без
'''''without''' doubt''       = ''несомненно''
''the knight '''without''' fear ~ and '''without''' reproach'' = ''рыцарь без страха и упрёка''


at last                       = наконец
at least                      = по крайней мере
at any rate                   = в любом случае
at this time                  = на этот раз
at full speed                 = на полной скорости




back @ backside %% <noun>     = спина, спинка, зад
back @ inside} %% <noun>      = изнанка, подкладка, ~ оборотная сторона
back @ sport} %% <noun>       = в спорте: защитник
back @ adj.} %% <adj.>        = задний, отдаленный, ~ обратный, ~ отсталый
back @ verb} %% <verb>        = поддерживать, подкреплять, ~ служить фоном
back @ adv.} %% <adv.>        = назад, обратно
back @ adv. Am.} {Am.} %% <adv.> = сзади, позади

''When do you expect her '''back'''?'' = ''Когда она должна вернуться?''
