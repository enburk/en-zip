basic ~ vocabulary

dozen          = дюжина
hundred        = сто, сотня
thousand       = тысяча
one and a half = полтора

weight         = вес, взвешивать
light @ weight = лёгкий
heavy @ weight = тяжёлый
---
size           = размер
length         = длина
long           = длинный
short          = короткий
: short pasta
: long pasta

height         = высота
high           = высокий
low            = низкий

thickness      = толщина
thick          = толстый
thin           = тонкий

depth          = глубина
deep           = глубокий
shallow        = мелкий
---
wide           = широкий
narrow         = узкий
: Open wide and say “ahh”.

hour           = час
hourglass      = песочные часы

===
volume @ space = объём
volume @ sound = громкость
volume @ book  = том
