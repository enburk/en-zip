basic ~ vocabulary

town           = город, городок, ~ центр города
suburb         = пригород /// Latin
village        = деревня, поселок

palace         = дворец, особняк, ~ официальная резиденция

garden {n}     = сад, огород /// French
bench @ park   = скамейка, скамья

floor @ house {n} = пол
roof @ house {n} = крыша
window @ house {n} = окно
---
bed @ home {n} = кровать, постель
bedroom        = спальня, ~ гостиничный номер

wallpaper      = обои

===
place @ area   = место, помещать
place @ town   = город, местечко, ~ населенный пункт
place @ home   = дом, жилище
place @ square = полщадь
place @ role   = должность, место
===
