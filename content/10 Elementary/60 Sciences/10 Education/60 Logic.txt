logic            = логика

syllogism        = силлогизм %% «Все люди смертны, Сократ — человек; следовательно, Сократ смертен». /// (заключение на основе двух или более посылок)

sophism          = софизм %% «Что ты не терял, то имеешь. Рога ты не терял. Значит, у тебя рога». /// (фактически ложное сложное рассуждение, уловка)

aporia           = апо́рия/апори́я                                                       /// (логически верное утверждение, противоречащее реальности)

Zeno's paradoxes = парадоксы/апории Зенона

antinomy         = антиномия %% «То, что я сейчас говорю, является ложью».             /// Русская вики называет парадокс лжеца апорией

tautology        = тавтология %% тождественно истинное высказывание

deduction        = вывод, умозаключение, ~ дедукция %% метод рассуждения от общего к частному /// вычет

induction        = индукция %% метод рассуждения от частного к общему

abduction        = абдукция %% познавательная процедура принятия гипотез ~~ Метод Шерлока Холмса скорее абдуктивный, а не дедуктивный.
